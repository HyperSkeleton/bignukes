-- Scripts:

local function IAOZU_fake_script() -- Background.LocalScript 
	local script = Instance.new('LocalScript', Background)

	script.Parent.Selectable = true
	script.Parent.Active = true
	local dragging
	local dragInput
	local dragStart
	local startPos
	
	local function update(input)
		local delta = input.Position - dragStart
		script.Parent:TweenPosition(UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y),'Out','Quint',.15,true)
	end
	script.Parent.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 then
			dragging = true
			dragStart = input.Position
			startPos = script.Parent.Position
			
			repeat wait() until input.UserInputState == Enum.UserInputState.End
			dragging = false
		end
	end)
	script.Parent.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement then
			dragInput = input
		end
	end)
	game:GetService("UserInputService").InputChanged:Connect(function(input)
		if input == dragInput and dragging then
			update(input)
		end
	end)
	
	p = game:GetService("Players").LocalPlayer
	mouse = p:GetMouse()
	openedGui = true
	mouse.KeyDown:connect(function(key)
	        if key == "=" then
	    	if openedGui == true then
			openedGui = false
			script.Parent.Parent.Enabled = false
		elseif openedGui == false then
			openedGui = true
			script.Parent.Parent.Enabled = true
		end
	end
end)
	
end
coroutine.wrap(IAOZU_fake_script)()
